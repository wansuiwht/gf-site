func IsGrpcErrorNotFound(err error) bool {
  	if err != nil {
  		for e := err; e != nil; e = gerror.Unwrap(e) {
  			if s, ok := status.FromError(e); ok && s != nil && s.Code() == codes.NotFound {
  				return true
  			}
  		}
  	}
  	return false
}
