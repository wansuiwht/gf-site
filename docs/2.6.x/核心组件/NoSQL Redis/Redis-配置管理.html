package main

    "fmt"

    "github.com/gogf/gf/v2/database/gredis"
    "github.com/gogf/gf/v2/frame/g"

)

var (
    config = gredis.Config{
        Address: "127.0.0.1:6379",

        g.Log().Fatal(ctx, err)
    }
    value, err := g.Redis(group).Get(ctx, "key")
    if err != nil {
        g.Log().Fatal(ctx, err)
    }
    fmt.Println(value.String())
}
