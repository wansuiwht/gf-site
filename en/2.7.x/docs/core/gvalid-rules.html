          Regex2 string `v:"not-regex:\\d{4}"`
      }
      var (
          ctx = context.Background()
          req = BizReq{
              Regex1: "1234",
              Regex2: "1234",
          }
      )
      if err := g.Validator().Data(req).Run(ctx); err != nil {
          fmt.Print(gstr.Join(err.Strings(), "\n"))
      }

      // Output:
      // The Regex2 value `1234` should not be in regex of: \d{4}
}
