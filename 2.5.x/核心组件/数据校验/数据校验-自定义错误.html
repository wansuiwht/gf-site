func main() {
    type User struct {
        Name string `v:"required#ReuiredUserName"`
        Type int    `v:"required#ReuiredUserType"`

    var (
        ctx  = gctx.New()
        data = g.Map{
            "name": "john",
        }

        ctxCh = gi18n.WithLanguage(ctx, "zh-CN")

    if err := gconv.Scan(data, &user); err != nil {
        panic(err)
